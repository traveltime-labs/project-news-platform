## 3. 基底類別（crawlers/base.py）

```python
from dataclasses import dataclass, field
from typing import Optional
import hashlib

@dataclass
class NewsItem:
    source: str
    title: str
    url: str
    summary: str
    category: str
    published_at: str        # ISO 8601
    url_hash: str = ''
    title_zh: Optional[str] = None
    summary_zh: Optional[str] = None
    is_sent: bool = False

    def __post_init__(self):
        self.url_hash = hashlib.md5(self.url.encode()).hexdigest()


class BaseCrawler:
    source_name: str = ''
    category: str = ''

    def fetch(self) -> list[NewsItem]:
        raise NotImplementedError

    def safe_fetch(self) -> list[NewsItem]:
        try:
            return self.fetch()
        except Exception as e:
            print(f'[{self.source_name}] 抓取失敗：{e}')
            return []
```

## 4. RSS 通用爬蟲（crawlers/rss.py）

Google News RSS 的連結是跳轉網址，需要額外解析出真正的原文網址。

```python
import feedparser
import requests
from datetime import datetime
from urllib.parse import urlparse, parse_qs
from .base import BaseCrawler, NewsItem

def resolve_google_news_url(google_url: str) -> str:
    """
    解析 Google News 跳轉連結，取得原文網址。
    失敗時回傳原本的 Google 跳轉網址。
    """
    try:
        resp = requests.get(google_url, allow_redirects=True, timeout=5)
        return resp.url
    except Exception:
        return google_url


class RssCrawler(BaseCrawler):
    def __init__(
        self,
        source_name: str,
        feed_url: str,
        category: str,
        is_google_news: bool = False,
    ):
        self.source_name = source_name
        self.feed_url = feed_url
        self.category = category
        self.is_google_news = is_google_news  # 是否需要解析跳轉連結

    def fetch(self) -> list[NewsItem]:
        feed = feedparser.parse(self.feed_url)
        items = []

        for entry in feed.entries[:20]:
            url = entry.get('link', '')

            # Google News 跳轉連結解析
            if self.is_google_news and url:
                url = resolve_google_news_url(url)

            published_at = entry.get('published', datetime.utcnow().isoformat())
            summary = entry.get('summary', '')[:500]

            # 來源名稱：Google News RSS 的 source 欄位有原始媒體名
            source = self.source_name
            if self.is_google_news and hasattr(entry, 'source'):
                source = entry.source.get('title', self.source_name)

            items.append(NewsItem(
                source=source,
                title=entry.get('title', '').strip(),
                url=url,
                summary=summary,
                category=self.category,
                published_at=published_at,
            ))

        return items
```

## 5. Google Sheets 操作（storage/sheets.py）

```python
import gspread
from google.oauth2.service_account import Credentials
import os

SCOPES = ['https://www.googleapis.com/auth/spreadsheets']

def get_client():
    creds = Credentials.from_service_account_info(
        {
            'type': 'service_account',
            'client_email': os.environ['GOOGLE_SERVICE_ACCOUNT_EMAIL'],
            'private_key': os.environ['GOOGLE_PRIVATE_KEY'].replace('\\n', '\n'),
            'token_uri': 'https://oauth2.googleapis.com/token',
        },
        scopes=SCOPES,
    )
    return gspread.authorize(creds)

def get_sheet(sheet_name: str):
    client = get_client()
    spreadsheet = client.open_by_key(os.environ['GOOGLE_SHEET_ID'])
    return spreadsheet.worksheet(sheet_name)


def get_sources() -> list[dict]:
    """從 Settings 工作表取得啟用的來源清單"""
    sheet = get_sheet('Settings')
    records = sheet.get_all_records()
    return [r for r in records if str(r.get('enabled', '')).upper() == 'TRUE']


def get_existing_hashes() -> set[str]:
    """取得所有已存在的 url_hash，用於去重"""
    sheet = get_sheet('News')
    records = sheet.get_all_records()
    return {row['url_hash'] for row in records if row.get('url_hash')}


def batch_write(items: list[dict]):
    """整批寫入，避免逐筆 API 呼叫觸發 429"""
    if not items:
        print('沒有新資料需要寫入')
        return

    sheet = get_sheet('News')
    rows = [
        [
            item['published_at'],
            item['source'],
            item['title'],
            item['url'],
            item['url_hash'],
            item['summary'],
            item['category'],
            '',       # title_zh
            '',       # summary_zh
            'FALSE',  # is_sent
        ]
        for item in items
    ]
    sheet.append_rows(rows, value_input_option='RAW')
    print(f'寫入 {len(rows)} 筆資料')
```

## 6. 去重邏輯（utils/dedup.py）

```python
def filter_new_items(items: list, existing_hashes: set[str]) -> list:
    """過濾掉已存在的新聞，只回傳新的"""
    return [item for item in items if item.url_hash not in existing_hashes]
```

## 7. 心跳檢測（utils/heartbeat.py）

> 狀態存於 Google Sheets `Settings` 工作表的 `fail_count` 欄位，跨執行持久保存。

```python
from storage.sheets import get_sheet

FAIL_THRESHOLD = 3

def check(source_name: str, item_count: int, notify_fn):
    sheet = get_sheet('Settings')
    records = sheet.get_all_records()

    for i, row in enumerate(records):
        if row.get('name') != source_name:
            continue

        fail_count = int(row.get('fail_count', 0))

        if item_count == 0:
            fail_count += 1
            if fail_count >= FAIL_THRESHOLD:
                notify_fn(f'⚠️ [{source_name}] 連續 {fail_count} 次抓取結果為空，請確認來源狀態。')
                fail_count = 0  # 告警後重置
        else:
            fail_count = 0  # 成功後重置

        # 更新 fail_count 欄位（列索引從 2 開始，因為第 1 列是 header）
        header = sheet.row_values(1)
        col_idx = header.index('fail_count') + 1
        sheet.update_cell(i + 2, col_idx, fail_count)
        break
```

## 8. 主程式（main.py）

```python
from crawlers.rss import RssCrawler
from storage.sheets import get_sources, get_existing_hashes, batch_write
from utils.dedup import filter_new_items
from utils.heartbeat import check as heartbeat_check
from utils.telegram import send_message
from dataclasses import asdict

def build_crawler(source: dict) -> RssCrawler:
    """從 Settings 工作表的一列資料建立對應的爬蟲"""
    feed_url = source['feed_url']
    is_google_news = 'news.google.com' in feed_url

    return RssCrawler(
        source_name=source['name'],
        feed_url=feed_url,
        category=source['category'],
        is_google_news=is_google_news,
    )

def main():
    # 從 Settings 動態讀取啟用的來源
    sources = get_sources()
    if not sources:
        print('沒有啟用的來源')
        return

    existing_hashes = get_existing_hashes()
    all_new_items = []

    for source in sources:
        crawler = build_crawler(source)
        items = crawler.safe_fetch()

        # 心跳檢測
        heartbeat_check(
            source_name=crawler.source_name,
            item_count=len(items),
            notify_fn=send_message,
        )

        # 去重
        new_items = filter_new_items(items, existing_hashes)
        all_new_items.extend(new_items)

        # 更新本次已知的 hash
        existing_hashes.update(item.url_hash for item in new_items)

        print(f'[{crawler.source_name}] 抓取 {len(items)} 則，新增 {len(new_items)} 則')

    # 整批寫入
    batch_write([asdict(item) for item in all_new_items])
    print(f'完成：共新增 {len(all_new_items)} 則新聞')

if __name__ == '__main__':
    main()
```

## 10. Google Sheets 初始化腳本（setup_sheets.py）

**只需執行一次**，自動建立三個工作表、填入欄位名稱、寫入預設來源。

```python
import gspread
from google.oauth2.service_account import Credentials
import os

SCOPES = ['https://www.googleapis.com/auth/spreadsheets']

SOURCES = [
    {"name": "國際科技",     "feed_url": "https://news.google.com/rss/headlines/section/topic/TECHNOLOGY?hl=en-US&gl=US&ceid=US:en",         "category": "tech"},
    {"name": "國際財經",     "feed_url": "https://news.google.com/rss/headlines/section/topic/BUSINESS?hl=en-US&gl=US&ceid=US:en",           "category": "finance"},
    {"name": "世界局勢",     "feed_url": "https://news.google.com/rss/headlines/section/topic/WORLD?hl=en-US&gl=US&ceid=US:en",              "category": "world"},
    {"name": "台灣新聞",     "feed_url": "https://news.google.com/rss/headlines/section/geo/Taiwan?hl=zh-TW&gl=TW&ceid=TW:zh-Hant",         "category": "taiwan"},
    {"name": "UFO新聞",     "feed_url": "https://news.google.com/rss/search?q=UFO+UAP&hl=en-US&gl=US&ceid=US:en",                          "category": "ufo"},
    {"name": "Reddit UFOs", "feed_url": "https://www.reddit.com/r/UFOs/top/.rss?sort=top&t=day&limit=20",                                   "category": "ufo"},
    {"name": "NASA",        "feed_url": "https://www.nasa.gov/news-release/feed/",                                                          "category": "ufo"},
]

def get_client():
    creds = Credentials.from_service_account_info(
        {
            'type': 'service_account',
            'client_email': os.environ['GOOGLE_SERVICE_ACCOUNT_EMAIL'],
            'private_key': os.environ['GOOGLE_PRIVATE_KEY'].replace('\\n', '\n'),
            'token_uri': 'https://oauth2.googleapis.com/token',
        },
        scopes=SCOPES,
    )
    return gspread.authorize(creds)

def setup():
    client = get_client()
    spreadsheet = client.open_by_key(os.environ['GOOGLE_SHEET_ID'])
    existing = [ws.title for ws in spreadsheet.worksheets()]

    # ── News ──────────────────────────────────────
    if 'News' not in existing:
        spreadsheet.add_worksheet(title='News', rows=1000, cols=10)
        print('建立 News 工作表')
    news = spreadsheet.worksheet('News')
    if not news.row_values(1):
        news.append_row([
            'published_at', 'source', 'title', 'url',
            'url_hash', 'summary', 'category',
            'title_zh', 'summary_zh', 'is_sent'
        ])
        print('News 欄位建立完成')

    # ── Bookmarks ─────────────────────────────────
    if 'Bookmarks' not in existing:
        spreadsheet.add_worksheet(title='Bookmarks', rows=1000, cols=2)
        print('建立 Bookmarks 工作表')
    bookmarks = spreadsheet.worksheet('Bookmarks')
    if not bookmarks.row_values(1):
        bookmarks.append_row(['newsId', 'savedAt'])
        print('Bookmarks 欄位建立完成')

    # ── Settings ──────────────────────────────────
    if 'Settings' not in existing:
        spreadsheet.add_worksheet(title='Settings', rows=100, cols=7)
        print('建立 Settings 工作表')
    settings = spreadsheet.worksheet('Settings')
    if not settings.row_values(1):
        settings.append_row([
            'name', 'feed_url', 'category',
            'enabled', 'fail_count',
            'notify_enabled', 'notify_times'
        ])
        rows = [
            [s['name'], s['feed_url'], s['category'], 'TRUE', 0, 'TRUE', '08:00,20:00']
            for s in SOURCES
        ]
        settings.append_rows(rows)
        print(f'Settings 寫入 {len(rows)} 筆預設來源')

    print('\n✅ 初始化完成！')

if __name__ == '__main__':
    setup()
```

## 11. GitHub Actions 設定

### 爬蟲排程（crawler.yml）

```yaml
name: 新聞爬蟲

on:
  schedule:
    - cron: '0 0,12 * * *'   # 每天 UTC 00:00 / 12:00（台灣 08:00 / 20:00）
  workflow_dispatch:           # 支援手動觸發，方便測試

jobs:
  crawl:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - uses: actions/setup-python@v5
        with:
          python-version: '3.11'

      - name: 安裝套件
        run: pip install -r crawler/requirements.txt

      - name: 執行爬蟲
        working-directory: crawler
        env:
          GOOGLE_SERVICE_ACCOUNT_EMAIL: ${{ secrets.GOOGLE_SERVICE_ACCOUNT_EMAIL }}
          GOOGLE_PRIVATE_KEY: ${{ secrets.GOOGLE_PRIVATE_KEY }}
          GOOGLE_SHEET_ID: ${{ secrets.GOOGLE_SHEET_ID }}
          TELEGRAM_BOT_TOKEN: ${{ secrets.TELEGRAM_BOT_TOKEN }}
          TELEGRAM_CHAT_ID: ${{ secrets.TELEGRAM_CHAT_ID }}
        run: python main.py
```
